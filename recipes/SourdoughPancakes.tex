\begin{recipe}{Sourdough Pancakes}{8 Pancakes}{}
    \ingredient[190]{grams}{Flour}
    \ingredient[2]{tbsp}{Sugar}
    \ingredient[1/2]{tsp}{Salt}
    \ingredient[240]{grams}{Sourdough discard}
    \ingredient[2]{}{Eggs}
    \ingredient[1]{Cup}{Milk}
    \ingredient[3]{tbsp}{Melted Butter}
    Combine wet and dry ingredients and whisk. Cover and chill over night.

    \ingredient[2]{tsp}{Baking Powder}
    \ingredient[1]{tsp}{Baking Soda}
    Add baking soda and baking powder. Then cook. Should be bubbly.
\end{recipe}

\begin{recipe}{Bread Dipping Oil}{}{}
    \ingredient[1]{tbsp}{Oregano}
    \ingredient[1]{tbsp}{Basil}
    \ingredient[1]{tbsp}{Parsley}
    \ingredient[2]{tsp}{Onion Powder}
    \ingredient[2]{tsp}{Black Pepper}
    \ingredient[2]{tsp}{Red Pepper Flakes}
    \ingredient[1 1/2]{tsp}{Salt}
    \ingredient[1/2]{tsp}{Rosemary}
    \ingredient[1/2]{tsp}{Garlic Powder}
    Add oregano, basil, parsley, onion powder,
    black pepper, red pepper flakes, salt, rosemary,
    and garlic powder in bowl.

    \ingredient[1/2]{Cup}{Extra Virgin Olive Oil}
    Add 1 tbsp of dry mixture to 1/2 cup of olive oil.

    \ingredient[1]{tbsp}{Minced Garlic}
    \ingredient[2]{tsp}{Balsamic Vinegar}
    Add minced garlic to taste and a splash
    of balsamic vinegar.
\end{recipe}

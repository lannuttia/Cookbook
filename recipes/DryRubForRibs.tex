\begin{recipe}{Dry Rub for Ribs}{}{}
    \ingredient[3]{tbsp}{Brown Sugar}
    \ingredient[1 1/2]{tbsp}{Paprika}
    \ingredient[1 1/2]{tbsp}{Salt}
    \ingredient[1 1/2]{tbsp}{Black Pepper}
    \ingredient[1]{tsp}{Garlic Powder}
    Mix in a bowl and sprinkle over ribs. Let
    it rest from 1 --- 8 hours before cooking.
\end{recipe}

\begin{recipe}{Lime Rice}{}{}
    \ingredient[1]{Cup}{Jasmine Rice}
    Cook rice.
    \ingredient[1]{tsp}{Salt}
    \ingredient[3]{tbsp}{Chopped Cilantro}
    \ingredient[2]{tbsp}{Butter}
    \ingredient[1]{tbsp}{Lime Juice}
    Use fork to fluff rice then gently fold in cilantro,
    butter, salt, and lime juice into rice.
\end{recipe}

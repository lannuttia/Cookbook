\begin{recipe}{Bean Burritos}{}{}
    Preheat oven to 375 \degrees{}F.
    \ingredient[1]{Can}{Refried Beans}
    \ingredient[1]{Cup}{Salsa}
    \ingredient[1]{Cup}{Cooked Rice}
    \ingredient[2]{Cups}{Shredded Cheese}
    Mix beans, salsa, rice, and cheese in a bowl.
    \ingredient[12]{6 inch}{Tortillas}
    Add filling to tortillas. Bake for 20 minutes.

\end{recipe}

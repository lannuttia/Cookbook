\begin{recipe}{Deery Cheese Steaks}{4 -- 6 sandwiches}{30 minutes}
    \ingredient[1]{Package}{Deer Steaks}
    \ingredient[1/2]{Cup}{Beef Broth}
    \ingredient[1]{tsp}{Garlic Powder}
    \ingredient[1]{tsp}{Onion Powder}
    \ingredient[]{}{Salt}
    \ingredient[]{}{Pepper}
    \ingredient[1/4]{tsp}{Paprika}
    \ingredient[1]{tsp}{Worchestirshire Sauce}
    Marinnade deer in broth and spices for 2 hours. Deer needs sliced before it
    is put in the marinnade.
    \ingredient[2]{}{Bell Peppers}
    \ingredient[8]{Oz}{Mushrooms}
    \ingredient[2]{}{Onions}
    \ingredient[1]{tbsp}{Butter}
    Chop onions, bell peppers, and mushrooms and sautee. Pull deer out of
    marinnade and cook in skillet. Once cooked, assemble sandwiches and serve.

\end{recipe}

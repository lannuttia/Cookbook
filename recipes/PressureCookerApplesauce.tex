\begin{recipe}{Pressure Cooker Applesauce}{}{}
    \ingredient[8 -- 10]{}{Apples}
    Peel and core apples.

    \ingredient[1]{tsp}{Cinnamon}
    \ingredient[1]{tsp}{Vanilla Extract}
    \ingredient[1/2]{Cup}{Water}
    \ingredient[1/4]{Cup}{Sugar}
    Add ingredients to pressure cooker. Cook for 5 minutes.
    Once cooked, blend ingredients.

\end{recipe}

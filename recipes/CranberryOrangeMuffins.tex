\begin{recipe}{Cranberry Orange Muffins}{}{}
    Preheat oven to 350\degrees{} F.
    \ingredient[2]{Cups}{Whole Wheat Flour}
    \ingredient[1/3]{Cup}{Sugar}
    \ingredient[1/3]{Cup}{Brown Sugar}
    \ingredient[2]{tsp}{Baking Powder}
    \ingredient[1/2]{tsp}{Salt}
    \ingredient[1]{tsp}{Baking Soda}
    \ingredient[1/2]{tsp}{Cinnamon}
    In a large bowl, whisk flour, sugar, brown sugar,
    baking soda, baking powder, salt, and cinnamon.

    \ingredient[1/2]{Cup}{Milk}
    \ingredient[1/4]{Cup}{Orange Juice}
    \ingredient[2]{}{Eggs}
    \ingredient[2]{tsp}{Vanilla Extract}
    In a separate bowl, whisk milk, orange juice, eggs,
    and vanilla extract. Add wet ingredients to dry and
    stir until combined.
    
    \ingredient[1/2]{Cup}{Melted Butter}
    Add in melted butter and combine.

    \ingredient[1 1/4]{Cup}{Fresh or Frozen Cranberries}
    \ingredient[1]{}{Orange's Zest}
    Fold in cranberries and orange zest. Divide batter
    among twelve muffin cups. Put in tins.

    \ingredient[]{}{Optional Course Sugar}
    Optionally sprinkle with course sugar. Bake for 20
    --- 25 minutes.
\end{recipe}

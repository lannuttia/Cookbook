\begin{recipe}{Steak Marinade}{}{}
    \ingredient[1/3]{Cup}{Soy Sauce}
    \ingredient[1/2]{Cup}{Melted Butter}
    \ingredient[1/3]{Cup}{Lemon Juice}
    \ingredient[1/4]{Cup}{Worcestershire}
    \ingredient[1 1/2]{tbsp}{Garlic Powder}
    \ingredient[2]{tbsp}{Dried Basil}
    \ingredient[1 1/2]{tbsp}{Parsley}
    \ingredient[1]{tsp}{Black Pepper}
    \ingredient[1]{tsp}{Minced Garlic}
    Mix ingredients in a bowl. Allow up to 8 hours for
    steak to marinade.
\end{recipe}

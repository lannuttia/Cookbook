\begin{recipe}{Beef Kabob Marinade}{}{}
    \ingredient[1/4]{Cup}{Worcestershire}
    \ingredient[1/4]{Cup}{Soy Sauce}
    \ingredient[1/4]{Cup}{Olive Oil}
    \ingredient[1]{tbsp}{dijon mustard}
    \ingredient[1]{tbsp}{Minced Garlic}
    \ingredient[1]{tsp}{Dried Basil}
    \ingredient[1]{tsp}{Dried Parsley}
    \ingredient[1]{tsp}{Black Pepper}
    Mix all ingredients in a bowl. Save 3 tbsp for pouring over
    beef kabobs after cooking. Allow 1 --- 6 hours for beef to
    marinade.
\end{recipe}
